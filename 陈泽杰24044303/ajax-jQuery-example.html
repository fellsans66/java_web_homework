jQuery Ajax 异步

使用 jQuery 封装的 $.ajax() / $.get() / $.post() 方法，代码更简洁。

核心概念： Ajax 是 "Asynchronous JavaScript and XML" 的缩写。异步 意味着发送请求后不会阻塞页面，浏览器可以继续响应用户操作， 等服务器返回数据后再通过回调函数处理结果。所有请求均模拟与 https://jsonplaceholder.typicode.com 交互。
纯 Ajax vs jQuery Ajax 对比： 纯 Ajax 需要手动处理 XMLHttpRequest 的创建、状态监听、错误处理等，代码量较大； jQuery Ajax 封装了这些细节，提供了更简洁的 API（$.get, $.post, $.ajax）， 并自动处理了跨浏览器兼容性问题。但 jQuery 需要额外引入库文件。 
// ========== 纯 Ajax（XMLHttpRequest）==========

/* --- GET 请求 --- */
function pureGet() {
    var xhr = new XMLHttpRequest();
    var resultEl = document.getElementById('pureResult');
    resultEl.className = 'result-box loading';
    resultEl.textContent = '正在请求数据...';

    // xhr.readyState 状态码说明：
    // 0: UNSENT       - 请求未初始化
    // 1: OPENED       - open() 已调用
    // 2: HEADERS_RECEIVED - 已收到响应头
    // 3: LOADING      - 响应体下载中
    // 4: DONE         - 请求完成

    xhr.onreadystatechange = function () {
        if (xhr.readyState === 4) {           // 请求完成
            if (xhr.status === 200) {          // HTTP 200 表示成功
                var data = JSON.parse(xhr.responseText);
                resultEl.className = 'result-box success';
                resultEl.textContent = '状态: ' + xhr.status
                    + '\n数据: ' + JSON.stringify(data, null, 2);
            } else {
                resultEl.className = 'result-box error';
                resultEl.textContent = '错误! 状态码: ' + xhr.status;
            }
        }
    };

    // 网络错误处理（断网等情况）
    xhr.onerror = function () {
        resultEl.className = 'result-box error';
        resultEl.textContent = '网络请求失败!';
    };

    xhr.open('GET', 'https://jsonplaceholder.typicode.com/posts/1', true);
    // 第三个参数 true 表示异步，false 表示同步（会阻塞页面，不推荐）
    xhr.send();
}

/* --- POST 请求 --- */
function purePost() {
    var xhr = new XMLHttpRequest();
    var resultEl = document.getElementById('pureResult');
    resultEl.className = 'result-box loading';
    resultEl.textContent = '正在提交数据...';

    xhr.onreadystatechange = function () {
        if (xhr.readyState === 4) {
            if (xhr.status === 201) {  // 201 Created
                var data = JSON.parse(xhr.responseText);
                resultEl.className = 'result-box success';
                resultEl.textContent = '状态: ' + xhr.status
                    + '\n创建成功! ID: ' + data.id
                    + '\n完整响应: ' + JSON.stringify(data, null, 2);
            } else {
                resultEl.className = 'result-box error';
                resultEl.textContent = '错误! 状态码: ' + xhr.status;
            }
        }
    };

    xhr.open('POST', 'https://jsonplaceholder.typicode.com/posts', true);
    xhr.setRequestHeader('Content-Type', 'application/json');
    // 发送 JSON 格式的数据
    xhr.send(JSON.stringify({
        title: '纯 Ajax 示例',
        body: '这是通过 XMLHttpRequest 发送的数据',
        userId: 1
    }));
}

/* --- GET 请求带参数 --- */
function pureGetWithParams() {
    var xhr = new XMLHttpRequest();
    var resultEl = document.getElementById('pureResult');
    resultEl.className = 'result-box loading';
    resultEl.textContent = '正在请求数据...';

    xhr.onreadystatechange = function () {
        if (xhr.readyState === 4) {
            if (xhr.status === 200) {
                var data = JSON.parse(xhr.responseText);
                resultEl.className = 'result-box success';
                resultEl.textContent = '状态: ' + xhr.status
                    + '\n获取到 ' + data.length + ' 条数据'
                    + '\n第一条: ' + JSON.stringify(data[0], null, 2);
            } else {
                resultEl.className = 'result-box error';
                resultEl.textContent = '错误! 状态码: ' + xhr.status;
            }
        }
    };

    // 参数拼接在 URL 查询字符串中
    xhr.open('GET', 'https://jsonplaceholder.typicode.com/posts?userId=1&_limit=3', true);
    xhr.send();
}
// ========== jQuery Ajax ==========

/* --- $.get() 简化 GET 请求 --- */
function jqGet() {
    var resultEl = $('#jqResult');
    resultEl.removeClass().addClass('result-box loading').text('正在请求数据...');

    $.get('https://jsonplaceholder.typicode.com/posts/1', function (data) {
        // 成功回调 - 数据已自动解析为 JSON 对象
        resultEl.removeClass().addClass('result-box success');
        resultEl.text('状态: 200' +
            '\n数据: ' + JSON.stringify(data, null, 2));
    }).fail(function (jqXHR) {
        // 失败回调
        resultEl.removeClass().addClass('result-box error');
        resultEl.text('请求失败! 状态码: ' + jqXHR.status);
    });
}

/* --- $.post() 简化 POST 请求 --- */
function jqPost() {
    var resultEl = $('#jqResult');
    resultEl.removeClass().addClass('result-box loading').text('正在提交数据...');

    $.post('https://jsonplaceholder.typicode.com/posts', {
        title: 'jQuery Ajax 示例',
        body: '这是通过 $.post() 发送的数据',
        userId: 1
    }, function (data) {
        resultEl.removeClass().addClass('result-box success');
        resultEl.text('状态: 201' +
            '\n创建成功! ID: ' + data.id +
            '\n完整响应: ' + JSON.stringify(data, null, 2));
    }).fail(function (jqXHR) {
        resultEl.removeClass().addClass('result-box error');
        resultEl.text('请求失败! 状态码: ' + jqXHR.status);
    });
}

/* --- $.ajax() 完全自定义配置 --- */
function jqAjaxCustom() {
    var resultEl = $('#jqResult');
    resultEl.removeClass().addClass('result-box loading').text('正在请求数据...');

    $.ajax({
        url: 'https://jsonplaceholder.typicode.com/posts',
        type: 'GET',           // 请求方法
        data: { userId: 1, _limit: 3 },  // 查询参数
        dataType: 'json',      // 期望的返回数据类型
        timeout: 5000,         // 超时时间(毫秒)
        beforeSend: function () {
            // 发送前执行：可添加 loading 动画等
            console.log('请求即将发送...');
        },
        success: function (data, textStatus, jqXHR) {
            // 成功回调
            resultEl.removeClass().addClass('result-box success');
            resultEl.text('状态: ' + jqXHR.status +
                '\n请求状态: ' + textStatus +
                '\n获取到 ' + data.length + ' 条数据' +
                '\n第一条: ' + JSON.stringify(data[0], null, 2));
        },
        error: function (jqXHR, textStatus, errorThrown) {
            // 失败回调
            resultEl.removeClass().addClass('result-box error');
            resultEl.text('请求失败!\n状态: ' + textStatus +
                '\n错误信息: ' + errorThrown);
        },
        complete: function () {
            // 无论成功失败都会执行
            console.log('请求已结束');
        }
    });
}
